#### `lib/api/client.ts`
```typescript
import axios from 'axios'

const apiClient = axios.create({
  baseURL: process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8001/api/v1',
  headers: {
    'Content-Type': 'application/json',
  },
})

// 请求拦截器 - 添加 JWT Token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('access_token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器 - 处理错误
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      // Token 过期,跳转登录
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

export default apiClient
```

#### `app/workflows/software_development.py`
```python
from langgraph.graph import StateGraph, END
from app.workflows.state import AgentState
from app.agents.coordinator import CoordinatorAgent
from app.agents.planner import PlannerAgent
from app.agents.backend_engineer import BackendEngineerAgent

def create_software_dev_workflow(project_id: str):
    """创建软件开发工作流"""
    
    # 初始化 Agents
    coordinator = CoordinatorAgent(project_id=project_id)
    planner = PlannerAgent(project_id=project_id)
    engineer = BackendEngineerAgent(project_id=project_id)
    
    # 创建工作流图
    workflow = StateGraph(AgentState)
    
    # 添加节点
    workflow.add_node("coordinator", coordinator.process)
    workflow.add_node("planner", planner.process)
    workflow.add_node("engineer", engineer.process)
    
    # 定义边
    workflow.add_edge("coordinator", "planner")
    workflow.add_edge("planner", "engineer")
    workflow.add_edge("engineer", END)
    
    # 设置入口点
    workflow.set_entry_point("coordinator")
    
    return workflow.compile()
```

```bash
#!/bin/bash

echo "Setting up AI Work OS..."

# 创建 .env 文件
if [ ! -f backend/.env ]; then
    cp backend/.env.example backend/.env
    echo "Created backend/.env"
fi

if [ ! -f frontend/.env.local ]; then
    cp frontend/.env.example frontend/.env.local
    echo "Created frontend/.env.local"
fi

# 启动 Docker 服务
docker-compose up -d postgres redis

# 等待数据库启动
echo "Waiting for database..."
sleep 5

# 运行数据库迁移
cd backend
poetry install
poetry run alembic upgrade head
cd ..

# 安装前端依赖
cd frontend
pnpm install
cd ..

echo "Setup complete!"
```

## 8. GitHub Actions 配置

```
.github/
└── workflows/
    ├── ci.yml               # 持续集成
    ├── cd.yml               # 持续部署
    └── test.yml             # 测试工作流
```

### 8.1 ci.yml

```yaml
name: CI

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  backend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
      
      - name: Install Poetry
        run: pip install poetry
      
      - name: Install dependencies
        working-directory: ./backend
        run: poetry install
      
      - name: Run tests
        working-directory: ./backend
        run: poetry run pytest
      
      - name: Run linter
        working-directory: ./backend
        run: poetry run ruff check .

  frontend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '20'
      
      - name: Install pnpm
        run: npm install -g pnpm
      
      - name: Install dependencies
        working-directory: ./frontend
        run: pnpm install
      
      - name: Run tests
        working-directory: ./frontend
        run: pnpm test
      
      - name: Run linter
        working-directory: ./frontend
        run: pnpm lint
      
      - name: Build
        working-directory: ./frontend
        run: pnpm build
```

## 10. 目录结构设计原则

### 10.1 模块化
- 每个功能模块独立目录
- 清晰的职责划分
- 易于维护和扩展

### 10.2 可测试性
- 测试文件与源文件对应
- 易于编写单元测试和集成测试

### 10.3 可扩展性
- 预留扩展空间
- 插件化设计
- 易于添加新功能

### 10.4 约定优于配置
- 遵循框架最佳实践
- 统一的命名规范
- 清晰的文件组织

## 11. 命名规范

### 11.1 文件命名
- **组件**: PascalCase (e.g., `ProjectCard.tsx`)
- **工具函数**: camelCase (e.g., `formatDate.ts`)
- **常量**: UPPER_SNAKE_CASE (e.g., `API_BASE_URL`)
- **类型**: PascalCase (e.g., `ProjectType`)

### 11.2 目录命名
- 使用小写和连字符 (e.g., `agent-tools`)
- 复数形式表示集合 (e.g., `components`, `services`)

### 11.3 变量命名
- **布尔值**: is/has/can 前缀 (e.g., `isLoading`, `hasError`)
- **函数**: 动词开头 (e.g., `fetchProjects`, `createAgent`)
- **常量**: 全大写 (e.g., `MAX_RETRIES`)
